//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Looks.SetEffectTo(BRIGHTNESS, 100);
    Looks.SwitchCostumeTo(costume1);
    Looks.SetEffectTo(GHOST, 100);
    Motion.GoToXY(0, 0);
    Forever
    {
        Looks.SwitchCostumeTo(Location);
        Looks.Show();
        If ((Not (List.ContainsItem(defeated, 33))) And (Location == 6))
        {
            Looks.Hide();
        }
        If ((Not (List.ContainsItem(Key Items, 52))) And (Location == 7))
        {
            Looks.Hide();
        }
        If ((enemies defeated on 16 < 5) And (Location == 16))
        {
            Looks.Hide();
        }
    }
}
